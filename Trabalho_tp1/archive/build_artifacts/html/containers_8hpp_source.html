178class ContainerReservas {
179private:
180 std::map<std::string, Reserva> reservas; // chave: codigo da reserva
181
182public:
183 // Converte Data "DD/MM/AAAA" ou "DD-MM-AAAA" em inteiro AAAAMMDD para comparacao
184 static int dataParaInt(const Data& d) {
185 std::string s = d.getValor();
186 std::string dd = s.substr(0,2);
187 std::string mm = s.substr(3,2);
188 std::string yyyy = s.substr(6,4);
189 return std::stoi(yyyy) * 10000 + std::stoi(mm) * 100 + std::stoi(dd);
190 }
191
192 bool adicionar(const Reserva& r) {
193 std::string cod = r.getCodigo().getValor();
194 if (reservas.count(cod)) return false;
195 if (!r.getHotel() || !r.getQuarto()) return false;
196 // Verificar disponibilidade antes de adicionar
197 Codigo codHotel = r.getHotel()->getCodigo();
198 Numero num = r.getQuarto()->getNumero();
199 if (!disponivel(codHotel, num, r.getCheckin(), r.getCheckout(), "")) return false;
200 reservas[cod] = r;
201 return true;
202 }
203
204 bool remover(const Codigo& cod) {
205 return reservas.erase(cod.getValor()) > 0;
206 }
207
208 std::vector<Reserva> listarPorEmail(const Email& e) const {
209 std::vector<Reserva> lista;
210 for (const auto& [codigo, r] : reservas) {
211 if (r.getHospede() && r.getHospede()->getEmail().getValor() == e.getValor()) lista.push_back(r);
212 }
213 return lista;
214 }
215
216 bool buscarReserva(const Codigo& cod, Reserva& out) const {
217 auto it = reservas.find(cod.getValor());
218 if (it == reservas.end()) return false;
219 out = it->second;
220 return true;
221 }
222
223 bool atualizarReserva(const Reserva& r) {
224 std::string cod = r.getCodigo().getValor();
225 if (!reservas.count(cod)) return false;
226 if (!r.getHotel() || !r.getQuarto()) return false;
227 if (!disponivel(r.getHotel()->getCodigo(), r.getQuarto()->getNumero(), r.getCheckin(), r.getCheckout(), cod)) return false;
228 reservas[cod] = r;
229 return true;
230 }
231
232 // Verifica disponibilidade (sem overlap) -- considera reserva existente opcionalmente excluida (codigoExcluido)
233 bool disponivel(const Codigo& codHotel, const Numero& num, const Data& chegada, const Data& partida, const std::string& codigoExcluido = "") const {
234 int inChegada = dataParaInt(chegada);
235 int inPartida = dataParaInt(partida);
236 for (const auto& [codigo, r] : reservas) {
237 if (!codigoExcluido.empty() && codigo == codigoExcluido) continue;
238 if (!r.getHotel() || !r.getQuarto()) continue;
239 if (r.getHotel()->getCodigo().getValor() != codHotel.getValor()) continue;
240 if (r.getQuarto()->getNumero().getValor() != num.getValor()) continue;
241 int a1 = dataParaInt(r.getCheckin());
242 int a2 = dataParaInt(r.getCheckout());
243 // overlap if chegada < a2 && a1 < partida
244 if (inChegada < a2 && a1 < inPartida) return false;
245 }
246 return true;
247 }
248
249 // Remove todas reservas de um hospede
250 void removerPorHospede(const Email& e) {
251 for (auto it = reservas.begin(); it != reservas.end(); ) {
252 if (it->second.getHospede() && it->second.getHospede()->getEmail().getValor() == e.getValor()) it = reservas.erase(it);
253 else ++it;
254 }
255 }

260 if (it->second.getHotel() && it->second.getHotel()->getCodigo().getValor() == codHotel.getValor()) it = reservas.erase(it);
261 else ++it;
262 }
263 }
